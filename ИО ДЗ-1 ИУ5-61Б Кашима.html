class DSU:
    def __init__(self, n):
        #    ( 0  n-1)     .
        # parent[i]    i.  parent[i] == i, i   .
        self.parent = list(range(n))
        # rank[i]     union (  ).
        #  -     .     
        #   ,    find  union.
        self.rank = [0] * n

    #  find    (path compression)
    #   () ,   i.
    #   ,       i     ,
    #     find   .
    def find(self, i):
        if self.parent[i] == i:
            return i
        #          i
        self.parent[i] = self.find(self.parent[i])
        return self.parent[i]

    #  union  
    #   ,   i  j.
    #    ,  i  j.
    #   ,   (          ).
    #   ,      ,     .
    #  True,    (.., i  j    ),  False.
    def union(self, i, j):
        root_i = self.find(i) #    i
        root_j = self.find(j) #    j

        #   ,   i  j     .
        #   (i, j)   ,    .
        if root_i != root_j:
            #   :          
            if self.rank[root_i] < self.rank[root_j]:
                self.parent[root_i] = root_j
            elif self.rank[root_i] > self.rank[root_j]:
                self.parent[root_j] = root_i
            else:
                #   ,    ,  root_i
                self.parent[root_j] = root_i
                #     
                self.rank[root_i] += 1
            return True #   
        return False #     ,    (  )

    #         
    #     DSU    .
    def get_partitions(self):
        #     .     .
        roots = [self.find(i) for i in range(len(self.parent))]

        #     . collections.defaultdict   .
        partitions_dict = collections.defaultdict(list)
        for i, root in enumerate(roots):
            partitions_dict[root].append(i)

        #     :
        # 1.     .
        # 2.          (   ).
        formatted_partitions = []
        #    (, _)    
        sorted_items = sorted(partitions_dict.items())

        for root, vertices in sorted_items:
             #     
             formatted_partitions.append(sorted(vertices))

        return formatted_partitions
