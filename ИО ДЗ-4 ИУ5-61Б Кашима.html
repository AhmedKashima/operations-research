def gcd(a, b):
    #

    return a

def simplify_inequality(a, b, c):
    #       
    #  ,     0,       .
    if a == 0 and b == 0 and c == 0:
        return 0, 0, 0 #  ,  
    #    
    common_divisor = abs(gcd(gcd(a, b), c))
    #    1,     
    if common_divisor > 1:
        return a // common_divisor, b // common_divisor, c // common_divisor
    #    
    return a, b, c
